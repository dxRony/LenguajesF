# Este es un programa sencillo para calcular el area de un triangulo

# Funcion para calcular el area del triangulo
deF calcular_area(base, altura):
    area = (base*altura)/2
    return area

try:
    base = float(input("Ingrese el valor de la base del triangulo: "))
    altura = float(input ("Ingrese el valor de la altura del triangulo: "))
    
    # Verificar que los valores ingresados sean validos (mayores a 0)
    if base >0 and altura > 0:
        area_triangulo = calcular_area(base, altura)
        print("El area del triangulo es: ", area_triangulo)
    else:
        print("Error: Los valores deben ser mayores a 0")
